13  /**
14   *
15   * Osa käyttöliittymälogiikkaa, kuuntelee kaupassa valittuja esineitä.
16   *
17   * @author Arttu
18   */
19  public class KauppaListener implements ActionListener {
20  
21      private JFrame frame;
22      private Hahmo hahmo;
23      private Esine esine;
24  
25      /**
26       *
27       * @param frame Kuunnellun KauppaGUIn frame.
28       * @param esine Mikä esine valittiin.
29       * @param hahmo Minkä hahmon esineisiin esine lisätään.
30       */
31      public KauppaListener(JFrame frame, Esine esine, Hahmo hahmo) {
32          this.frame = frame;
33          this.hahmo = hahmo;
34          this.esine = esine;
35      }
36  
37      @Override
38      public void actionPerformed(ActionEvent e) {
39          List<Esine> hahmonEsineet = this.hahmo.getEsineet();
40          hahmonEsineet.add(this.esine);
41          if (this.hahmo.getRaha() - this.esine.getHinta() >= 0) {
42              this.hahmo.setRaha(this.hahmo.getRaha() - this.esine.getHinta());
43              this.hahmo.setEsineet(hahmonEsineet);
44          }
45  
46          TaisteluGUI taistelu = null;
47  
48          taistelu = new TaisteluGUI(this.hahmo, 0);
49  
50          this.frame.dispose();
51          taistelu.run();
52          taistelu.paivitaEsine();
53      }
54  
55  }
