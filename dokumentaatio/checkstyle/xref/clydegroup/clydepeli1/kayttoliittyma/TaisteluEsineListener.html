3   import clydegroup.clydepeli1.kauppa.Esine;
4   import clydegroup.clydepeli1.logiikka.Taistelu;
5   import java.awt.event.ActionEvent;
6   import java.awt.event.ActionListener;
7   import javax.swing.JButton;
8   
9   /**
10   *
11   * Osa käyttöliittymän logiikkaa, kuuntelee taistelussa käytettyjä esineitä.
12   *
13   * @author Arttu
14   */
15  public class TaisteluEsineListener implements ActionListener {
16  
17      private Taistelu taistelu;
18      private Esine esine;
19  
20      /**
21       *
22       * @param taistelu
23       * @param esine
24       */
25      public TaisteluEsineListener(Taistelu taistelu, Esine esine) {
26          this.taistelu = taistelu;
27          this.esine = esine;
28      }
29  
30      @Override
31      public void actionPerformed(ActionEvent e) {
32          this.esine.kayta(this.taistelu.getPelaaja());
33          this.taistelu.getPelaaja().getEsineet().remove(e);
34  
35          this.taistelu.getGui().setEsineenKaytto(new JButton("Sinulla ei ole esinetä."));
36          this.taistelu.getGui().paivitaHP();
37      }
38  
39  }
